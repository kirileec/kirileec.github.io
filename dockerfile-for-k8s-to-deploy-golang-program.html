dockerfile for k8s to deploy golang program

PS: 安装 github.com/linxlib/k 可以直接通过 k init 生成这样的文件

# 先使用特定的go环境的镜像进行打包
FROM golang:1.18 as build
MAINTAINER "linx"

ENV GO111MODULE=on \
    CGO_ENABLED=0 \
    GOOS=linux \
    GOARCH=amd64 \
	GOPROXY="https://goproxy.cn" \
	GOPRIVATE="gitee.com"


RUN mkdir /src
# 安装编译工具 也可以用go直接打包
RUN go install github.com/linxlib/k@latest
WORKDIR /src

COPY . .
# 安装依赖
RUN go mod tidy
RUN k build

# 运行用的镜像
FROM ubuntu as prod
# 最好都放在单独的目录，不要放根目录，因为可能不方便进行挂载
# 特别是有好多个目录/文件需要挂载的时候
RUN mkdir /app
WORKDIR /app
# 设置镜像时区，防止时间错位
# 安装ssl相关证书，需要请求https时必须安装
RUN export DEBIAN_FRONTEND=noninteractive  \
    && apt-get update \
    && apt-get install -y tzdata \
    && ln -fs /usr/share/zoneinfo/Asia/Shanghai /etc/localtime \
    && apt-get -qq install -y --no-install-recommends ca-certificates curl \
    && dpkg-reconfigure --frontend noninteractive tzdata


# 从编译镜像中拷贝文件到运行镜像中
COPY --from=build /src/bin/1.0.0/linux_amd64/app_name .
COPY --from=build /src/bin/1.0.0/linux_amd64/gen.gob .
# 如果配置文件需要通过ConfigMap挂载，那这里就不需要它了
# COPY --from=build /src/bin/1.0.0/linux_amd64/config.toml  .
COPY --from=build /src/bin/1.0.0/linux_amd64/swagger.json .
# 这样链接一下比较好，不然下面经常提醒文件不存在什么的，明明文件就在那个目录
RUN ln -fs /app/app_name /usr/bin/app_name
RUN apt-get clean
EXPOSE 1509

CMD ["app_name"]
